// Contains returns true if list contains the given value, false otherwise.
func Contains(list []string, value string) bool {
        for _, listElement := range list {
                if value == listElement {
                        return true
                }
        }
        return false
}

func IsRedirectURIValid(redirectURI string, redirectURIs []string, allowHttp bool) bool {
        http := strings.HasPrefix(redirectURI, "http")
        https := http && len(redirectURI) >= 5 && redirectURI[4] == 's'
        protocolAllowed := https || http && allowHttp
        if !protocolAllowed {
                return false
        }
        for _, allowedRedirectURI := range redirectURIs {
                if redirectURI == allowedRedirectURI {
                        return true
                }
        }
        return false
}

type CodeMethod string

func IsCodeVerifierOrChallenge(codeVerifier string) bool {
        if len(codeVerifier) < 43 || len(codeVerifier) > 128 {
                return false
        }
        for _, c := range codeVerifier {
                if !isALPHA(c) && !isDIGIT(c) && c != '-' && c != '.' && c != '_' && c != '~' {
                        return false
                }
        }
        return true
}

func createCodeChallenge(codeVerifier []byte, method CodeMethod) ([]byte, error) {
        if !IsCodeVerifierOrChallenge(string(codeVerifier)) {
                return nil, errors.New("invalid code verifier")
        }
        switch method {
        case Plain:
                return codeVerifier, nil
        case S256:
                hash := sha256.Sum256(codeVerifier)
                encodedLen := base64.RawURLEncoding.EncodedLen(len(hash))
                codeChallenge := make([]byte, encodedLen)
                base64.RawURLEncoding.Encode(codeChallenge, hash[:])
                return codeChallenge, nil
        default:
                return nil, errors.New("not implemented")
        }
}

func isVSCHAR(char rune) bool {
        return char >= 0x20 && char <= 0x7e
}

func isNQCHAR(char rune) bool {
        return char == 0x21 ||
                (char >= 0x23 && char <= 0x5b) ||
                (char >= 0x5d && char <= 0x7e)
}

func isNQSCHAR(char rune) bool {
        return char == 0x20 ||
                char == 0x21 ||
                (char >= 0x23 && char <= 0x5b) ||
                (char >= 0x5d && char <= 0x7e)
}

func isUNICODECHARNOCRLF(char rune) bool {
        return char == 0x09 ||
                (char >= 0x20 && char <= 0x7e) ||
                (char >= 0x80 && char <= 0xd7ff) ||
                (char >= 0xe000 && char <= 0xfffd) ||
                (char >= 0x10000 && char <= 0x10ffff)
}

func isResponseType(responseType string) bool {
        var spaceAllowed bool
        for _, char := range responseType {
                if char == '_' || isDIGIT(char) || isALPHA(char) {
                        spaceAllowed = true
                        continue
                }
                if spaceAllowed && char == 0x20 {
                        spaceAllowed = false
                        continue
                }
                return false
        }
        return spaceAllowed
}

func ParseScope(scope string) ([]string, bool) {
        var spaceAllowed bool
        var scopes []string
        var last int
        for i, char := range scope {
                if isNQCHAR(char) {
                        spaceAllowed = true
                        continue
                }
                if spaceAllowed && char == 0x20 {
                        spaceAllowed = false
                        scopes = append(scopes, scope[last:i])
                        last = i + 1
                        continue
                }
                return nil, false
        }
        if !spaceAllowed {
                return nil, false
        }
        return append(scopes, scope[last:]), spaceAllowed
}

// authRequest contains all parameters of the authorization request
// sent to the authorization endpoint.
type authRequest struct {
        parameters map[string]string
}

func parseAuthRequest(args map[string][]string) (*authRequest, error) {
        var authRequest = &authRequest{
                parameters: map[string]string{},
        }
        for key, value := range args {
                switch key {
                case
                        responseType,
                        scope,
                        state,
                        responseMode,
                        nonce,
                        display,
                        prompt,
                        maxAge,
                        uiLocales,
                        idTokenHint,
                        loginHint,
                        acrValues,
                        clientID,
                        codeChallenge,
                        codeChallengeMethod,
                        redirectURI:
                        if len(value) != 1 {
                                return nil, errors.New(invalidRequest)
                        }
                        authRequest.parameters[key] = value[0]
                }
        }
        return authRequest, nil
}

func AuthorizeEndpointHandler(w http.ResponseWriter, r *http.Request, sessionStore dal.SessionStore, db *gorm.DB) {
        if r.Method == http.MethodPost || r.Method == http.MethodGet {
                if err := r.ParseForm(); err != nil {
                        handleBadRequest(w, invalidRequest)
                        return
                }
        } else {
                // HTTP DELETE, PATCH, PUT
                handleNotAllowed(w)
                return
        }
        request, err := parseAuthRequest(r.Form)
        if err != nil {
                handleBadRequest(w, invalidRequest)
                return
        }
        clientID := request.parameters[clientID]
        if !logic.IsClientID(clientID) {
                handleBadRequest(w, invalidClient)
                return
        }
        redirectURI := request.parameters[redirectURI]
        if !logic.IsRedirectURI(redirectURI) {
                handleBadRequest(w, invalidRedirectURI)
                return
        }
        client, err := clients.GetClient(clientID)
        if err != nil {
                handleBadRequest(w, invalidClient)
                return
        }
        // TODO compare client with authenticated client from HTTP BASIC
        if !logic.IsRedirectURIValid(redirectURI, client.RedirectURIs, client.AllowHTTP && !client.IsPublic) {
                handleBadRequest(w, invalidRedirectURI)
                return
        }
        stateValue, stateDefined := request.parameters[state]
        if stateDefined {
                if !logic.IsState(stateValue) {
                        r.URL.Query().Add(errParam, invalidRequest)
                        http.Redirect(w, r, redirectURI, http.StatusSeeOther)
                        return
                }
                r.URL.Query().Add(state, stateValue)
        }
        scopes, scopesDefined := logic.ParseScope(request.parameters[scope])
        if !scopesDefined || !logic.Contains(scopes, openID) {
                r.URL.Query().Add(errParam, invalidRequest)
                http.Redirect(w, r, redirectURI, http.StatusSeeOther)
                return
        }

        switch request.parameters[responseType] {
        case code:
                var (
                        challenge *string
                        ccMethod  logic.CodeMethod
                )

                if cc, ccDefined := request.parameters[codeChallenge]; ccDefined {
                        challenge = &cc
                        if !logic.IsCodeVerifierOrChallenge(cc) {
                                r.URL.Query().Add(errParam, invalidRequest)
                                http.Redirect(w, r, redirectURI, http.StatusSeeOther)
                                return
                        }
                        if method, ok := request.parameters[codeChallengeMethod]; ok {
                                ccMethod = logic.CodeMethod(method)
                        }
                } else {
                        if client.IsPublic {
                                // require PKCE but client is PUBLIC
                                r.URL.Query().Add(errParam, invalidRequest)
                                http.Redirect(w, r, redirectURI, http.StatusSeeOther)
                                return
                        }
                }

                fmt.Printf("%v %v\n", challenge, ccMethod)
                // TODO https://openid.net/specs/openid-connect-core-1_0.html#AuthRequestValidation (4)
                // TODO additional errors https://openid.net/specs/openid-connect-core-1_0.html#AuthError
                if request.parameters[prompt] == "login" {
                        http.SetCookie(w, &http.Cookie{
                                Name:  authCookie,
                                Value: "",
                        })
                        http.Redirect(w, r, LoginEndpoint, http.StatusSeeOther)
                        return
                }
                _, userIsAuthenticated := getSignedCookie(r, authCookie)
                if request.parameters[prompt] == "none" {
                        if !userIsAuthenticated {
                                r.URL.Query().Add(errParam, loginRequired)
                                http.Redirect(w, r, redirectURI, http.StatusSeeOther)
                                return
                        }
                }
                if !userIsAuthenticated {
                        // save auth request params to cookie
                        http.Redirect(w, r, LoginEndpoint, http.StatusSeeOther)
                        return
                }
        default:
                r.URL.Query().Add(errParam, unsupportedResponseType)
                http.Redirect(w, r, redirectURI, http.StatusSeeOther)
        }
}

// newSignedCookie returns the cookie with key cookieKey and the value given by message.
// It signs the message, encodes it and sets the signed encoded value on the cookie
func newSignedCookie(cookieKey string, message string) (*http.Cookie, error) {
        signedMessage, err := signMessage(message, globalPrivateKey)
        if err != nil {
                return nil, err
        }
        return &http.Cookie{
                Name:     cookieKey,
                Secure:   true,
                HttpOnly: true,
                SameSite: http.SameSiteStrictMode,
                Value:    string(encode(signedMessage)),
        }, nil
}

// getSignedCookie returns the cookie value as a slice of bytes.
// It is the counterpart of the newSignedCookie function
// It returns nil, false if it cannot decode the cookie or if the cookie is not set on the request.
func getSignedCookie(r *http.Request, cookieKey string) ([]byte, bool) {
        cookie, err := r.Cookie(cookieKey)
        if err != nil {
                return nil, false
        }
        decode, err := decode([]byte(cookie.Value))
        if err != nil {
                return nil, false
        }
        if verifyMessage(decode, globalPublicKey) {
                return decode.message, true
        }
        return nil, false
}

import (
        "bufio"
        "bytes"
        "crypto/ecdsa"
        "crypto/elliptic"
        "crypto/rand"
        "crypto/sha256"
        "encoding/base32"
        "math/big"
)

type (
        // signedMessage holds a message and a signature
        signedMessage struct {
                // message is the actual message
                message []byte
                // signature is the signature of the message
                signature signature
        }

        // signature is a signature created by the ECDSA algorithm
        signature struct {
                r *big.Int
                s *big.Int
        }
)

func encode(message *signedMessage) []byte {
        buf := new(bytes.Buffer)
        encoder := base32.NewEncoder(base32.StdEncoding, buf)
        if message.signature.r != nil {
                encoder.Write(message.signature.r.Bytes())
        }
        encoder.Write([]byte{'\n'})
        if message.signature.s != nil {
                encoder.Write(message.signature.s.Bytes())
        }
        encoder.Write([]byte{'\n'})
        encoder.Write(message.message)
        encoder.Close()
        return buf.Bytes()
}

func decode(message []byte) (*signedMessage, error) {
        decoder := base32.NewDecoder(base32.StdEncoding, bytes.NewReader(message))
        scanner := bufio.NewScanner(decoder)
        scanner.Scan()
        var r *big.Int
        var s *big.Int
        if len(scanner.Bytes()) > 0 {
                r = &big.Int{}
                r = r.SetBytes(scanner.Bytes())
        }
        scanner.Scan()
        if len(scanner.Bytes()) > 0 {
                s = &big.Int{}
                s = s.SetBytes(scanner.Bytes())
        }
        scanner.Scan()
        decoded := scanner.Bytes()
        if scanner.Err() != nil {
                return nil, scanner.Err()
        }
        return &signedMessage{
                message: decoded,
                signature: signature{
                        r: r,
                        s: s,
                },
        }, nil
}

func getPublicKey(private *ecdsa.PrivateKey) *ecdsa.PublicKey {
        return &ecdsa.PublicKey{
                X:     private.X,
                Y:     private.Y,
                Curve: private.Curve,
        }
}

func generatePrivateKey() (*ecdsa.PrivateKey, error) {
        private, err := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
        if err != nil {
                return nil, err
        }
        return private, nil
}

func verifyMessage(message *signedMessage, key *ecdsa.PublicKey) bool {
        if message == nil || key == nil || message.signature.r == nil || message.signature.s == nil {
                return false
        }
        hash := sha256.Sum256(message.message)
        return ecdsa.Verify(key, hash[:], message.signature.r, message.signature.s)
}

func signMessage(message string, key *ecdsa.PrivateKey) (*signedMessage, error) {
        hash := sha256.Sum256([]byte(message))
        r, s, err := ecdsa.Sign(rand.Reader, key, hash[:])
        if err != nil {
                return nil, err
        }
        return &signedMessage{
                message: []byte(message),
                signature: signature{
                        r: r,
                        s: s,
                },
        }, nil
}
